var addQuotesToEachString = function (content) {
  var depth = 0;
  var objectStart = undefined;
  var objectEnd = undefined;
  var removedObjects = [];
  for (var i = 0; i < content.length; i++) {
    if (content[i] == '{') {
      if (depth == 0) objectStart = i;
      depth++;
    } else if (content[i] == '}') {
      depth--;
